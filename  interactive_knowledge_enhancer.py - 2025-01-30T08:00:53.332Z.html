import streamlit as st import requests import random # Define topics and their respective content TOPICS = { 'Science': { 'content': 'Science is the pursuit and application of knowledge and understanding of the natural and social world following a systematic methodology based on evidence.', 'quiz': [ {'question': 'What is the chemical symbol for water?', 'options': ['H2O', 'CO2', 'NaCl', 'O2'], 'answer': 'H2O'}, {'question': 'Who discovered penicillin?', 'options': ['Alexander Fleming', 'Marie Curie', 'Isaac Newton', 'Albert Einstein'], 'answer': 'Alexander Fleming'} ] }, 'Technology': { 'content': 'Technology refers to methods, systems, and devices which are the result of scientific knowledge being used for practical purposes.', 'quiz': [ {'question': 'What does CPU stand for?', 'options': ['Central Processing Unit', 'Computer Processing Unit', 'Central Processor Unit', 'Computer Processor Unit'], 'answer': 'Central Processing Unit'}, {'question': 'Which company developed the Python programming language?', 'options': ['Google', 'Python Software Foundation', 'Microsoft', 'Guido van Rossum'], 'answer': 'Guido van Rossum'} ] }, 'History': { 'content': 'History is the study of the past, particularly how it relates to humans.', 'quiz': [ {'question': 'Who was the first President of the United States?', 'options': ['Thomas Jefferson', 'George Washington', 'Abraham Lincoln', 'John Adams'], 'answer': 'George Washington'}, {'question': 'In which year did World War II end?', 'options': ['1945', '1939', '1941', '1943'], 'answer': '1945'} ] }, 'General Knowledge': { 'content': 'General knowledge is information that has been accumulated over time through various mediums.', 'quiz': [ {'question': 'What is the capital of France?', 'options': ['Paris', 'London', 'Berlin', 'Madrid'], 'answer': 'Paris'}, {'question': 'How many continents are there?', 'options': ['5', '6', '7', '8'], 'answer': '7'} ] } } # Function to display content and quiz for a selected topic def display_topic(topic): st.write(f'## {topic}') st.write(TOPICS[topic]['content']) st.write('### Quiz') for i, q in enumerate(TOPICS[topic]['quiz']): st.write(f'{i+1}. {q['question']}') answer = st.radio(f'Options for question {i+1}', q['options'], key=f'{topic}_{i}') if answer == q['answer']: st.success('Correct!') else: st.error(f'Wrong! The correct answer is {q['answer']}') # Main function to run the Streamlit app def main(): st.title('Interactive Knowledge Enhancer') topic = st.sidebar.selectbox('Choose a topic', list(TOPICS.keys())) display_topic(topic) if __name__ == '__main__': main()
