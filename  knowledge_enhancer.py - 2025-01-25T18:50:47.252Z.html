import requests import random # Define the topics and their corresponding API endpoints or data TOPICS = { 'science': 'https://science-facts-api.com/random', 'technology': 'https://tech-facts-api.com/random', 'history': 'https://history-facts-api.com/random', 'general knowledge': 'https://general-knowledge-facts-api.com/random' } # Function to fetch a random fact from the selected topic def fetch_fact(topic): response = requests.get(TOPICS[topic]) if response.status_code == 200: return response.json()['fact'] else: return "Failed to fetch a fact. Please try again." # Function to display the menu and get user's choice def display_menu(): print("Welcome to Knowledge Enhancer!") print("Please select a topic to learn more about:") for i, topic in enumerate(TOPICS.keys(), 1): print(f"{i}. {topic.title()}") choice = int(input("Enter the number of your choice: ")) return list(TOPICS.keys())[choice-1] # Function to conduct a quiz on the selected topic def conduct_quiz(topic): print(f"\nLet's test your knowledge on {topic.title()}!") # Placeholder for quiz questions and answers quiz = { 'question': 'What is the capital of France?', 'options': ['A. Paris', 'B. London', 'C. Berlin', 'D. Madrid'], 'answer': 'A' } print(quiz['question']) for option in quiz['options']: print(option) user_answer = input("Your answer (A/B/C/D): ").upper() if user_answer == quiz['answer']: print("Correct!") else: print(f"Sorry, the correct answer is {quiz['answer']}.") # Main function to run the program def main(): topic = display_menu() fact = fetch_fact(topic) print(f"\nHere's a random fact about {topic.title()}:\n{fact}") conduct_quiz(topic) if __name__ == "__main__": main()
