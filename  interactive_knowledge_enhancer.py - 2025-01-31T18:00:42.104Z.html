import streamlit as st import requests import random # Function to fetch random fact from an API def fetch_random_fact(topic): url = f'https://api.example.com/facts?topic={topic}' response = requests.get(url) if response.status_code == 200: return response.json()['fact'] else: return 'Failed to fetch a fact. Please try again.' # Function to display a quiz def display_quiz(questions): score = 0 for i, q in enumerate(questions): st.write(f'Question {i+1}: {q['question']}') answer = st.radio('Choose the correct answer:', q['options'], key=f'q{i}') if answer == q['answer']: score += 1 st.write('Correct!') else: st.write(f'Wrong! The correct answer is {q['answer']}') st.write(f'Your final score is {score}/{len(questions)}') # Main function def main(): st.title('Interactive Knowledge Enhancer') topics = ['Science', 'Technology', 'History', 'General Knowledge'] choice = st.sidebar.selectbox('Choose a topic:', topics) if st.sidebar.button('Learn a Random Fact'): fact = fetch_random_fact(choice) st.write(f'**Random Fact:** {fact}') if st.sidebar.button('Take a Quiz'): # Example quiz questions questions = [ {'question': 'What is the capital of France?', 'options': ['Paris', 'London', 'Berlin', 'Madrid'], 'answer': 'Paris'}, {'question': 'Who invented the telephone?', 'options': ['Thomas Edison', 'Alexander Graham Bell', 'Nikola Tesla', 'Guglielmo Marconi'], 'answer': 'Alexander Graham Bell'}, {'question': 'What is the smallest unit of life?', 'options': ['Cell', 'Atom', 'Molecule', 'Organ'], 'answer': 'Cell'} ] display_quiz(questions) if __name__ == '__main__': main()
